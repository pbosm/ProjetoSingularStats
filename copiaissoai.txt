package metodofuncao;

import java.util.Scanner;

public class MetodoFuncao {

    public static void main(String[] args) {

        Scanner entrada = new Scanner(System.in);
               
        System.out.print("Informe seu nome: ");
        String nome = entrada.next();  //quando for string não precisa detalhar dps do next
       
        System.out.print("Informe sua idade: ");
        int idade = entrada.nextInt();
       
        System.out.print("Informe seu peso: ");
        float peso = entrada.nextFloat();
       
        System.out.print("Informe sua altura: ");
        float altura = entrada.nextFloat();
       
         
        float imc = calcularImc(peso, altura); //ENTRADA DA FUNÇÃO, FUNÇÃO DENTRO DO CALCULO
       
        //ENTRADA DA FUNÇÃO, para imprimir dados, essa não precisa criar uma variável pois a função lá me baixo é void
        imprimirDados(nome, idade, peso, altura, falarPeso(imc), calcularImc(peso, altura), verificarMaior(idade)); //e dentro dessa função a 3 função (falarPeso, calcularImc e verificarMaior)
           
        //ENTRADA DA FUNÇÃO, para verificar se o usuário é maior ou menor de idade
        //String resposta = verificarMaior(idade);
        //System.out.println(resposta);
       
    }
   
    public static String verificarMaior(int idade) {  //CONCLUSÃO DA FUNÇÃO, verificar se é maior ou menor de idade
       String resposta = (idade >= 18) ? "Usuário maior de idade" : "Usuário menor de idade";
       return resposta;
    }
   
    public static float calcularImc(float peso, float altura) { //CONCLUSÃO DA FUNÇÃO, calcular o IMC do usuário
       float imc = peso / (altura*altura);
       return imc;
    }
   
    public static String falarPeso(float imc) {  //CONCLUSÃO DA FUNÇÃO, classificar se o qual estado do usuário
        if(imc < 18.5) {
         return("Usuário está abaixo do peso"); }
        else if(imc < 25.0) {
         return("Usuário está no peso normal"); }
        else if(imc < 30.0) {
         return("Usuário está em sobrepeso"); }
        else if(imc < 40) {
         return("Está em obesidade"); }
        else {
         return("Obesidade grave"); }
    }
   
    public static void imprimirDados(String nome, int idade, float peso, float altura, String falarPeso, float calcularImc, String verificarMaior){ //CONCLUSÃO DA FUNÇÃO, imprimir esses dados no console
        System.out.println("Nome: " + nome);
        System.out.println("idade: " + idade);
        System.out.println("peso: " + peso);
        System.out.println("altura: " + altura);
        System.out.println(verificarMaior);
        System.out.println("Calculo do IMC: " + calcularImc);
        System.out.println("Classificação do IMC: " + falarPeso);
    }    
   
}
